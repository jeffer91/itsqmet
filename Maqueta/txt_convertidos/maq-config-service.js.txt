// Archivo: maq-config-service.js
// Utilidad: Genera la estructura visual del menú usando datos automáticos.
(function (window) {
  "use strict";

  var registry = window.MAQ_MODULOS_REGISTRY;

  function obtenerConfigEfectiva() {
    var autoData = window.MAQ_AUTODATA || [];

    // --- 1. MENÚ FIJO (Izquierda) ---
    var itemsFijos = [
      { moduloId: "carga_excel",      etiqueta: "Carga",        orden: 10 },
      { moduloId: "tabla_principal",  etiqueta: "Tabla",        orden: 20 },
      { moduloId: "ficha_estudiante", etiqueta: "Ficha",        orden: 30 },
      { moduloId: "stat_main",        etiqueta: "Estadísticas", orden: 40 },
      { moduloId: "modulo_reporte",   etiqueta: "Reporte",      orden: 999 } // Al final
    ];

    var itemsMenuCalculados = [];

    // Agregar Fijos
    itemsFijos.forEach(function(fijo) {
      itemsMenuCalculados.push({
        id: "menu_" + fijo.moduloId,
        tipo: "modulo",
        etiqueta: fijo.etiqueta,
        moduloId: fijo.moduloId,
        orden: fijo.orden
      });
    });

    // --- 2. MENÚ DINÁMICO (Centro/Derecha) ---
    // Usamos un offset de orden para que salgan después de los fijos
    var ordenBase = 50; 

    autoData.forEach(function(item, idx) {
      var ordenActual = ordenBase + idx;

      if (item.tipo === 'modulo') {
        itemsMenuCalculados.push({
          id: "menu_" + item.id,
          tipo: "modulo",
          etiqueta: item.etiqueta,
          moduloId: item.id,
          orden: ordenActual
        });
      } else if (item.tipo === 'grupo') {
        // Convertir hijos
        var hijosMenu = item.hijos.map(function(h) {
          return {
            id: "menu_" + h.id,
            etiqueta: h.etiqueta,
            moduloId: h.id
          };
        });

        itemsMenuCalculados.push({
          id: "grp_" + item.id,
          tipo: "grupo",
          etiqueta: item.etiqueta,
          hijos: hijosMenu,
          orden: ordenActual
        });
      }
    });

    // Ordenar final
    itemsMenuCalculados.sort(function(a, b) {
      return a.orden - b.orden;
    });

    return Promise.resolve({ itemsMenuCalculados: itemsMenuCalculados });
  }

  function construirItemsMenu(config) {
    return config.itemsMenuCalculados || [];
  }

  window.MAQ_CONFIG_SERVICE = {
    obtenerConfigEfectiva: obtenerConfigEfectiva,
    construirItemsMenu: construirItemsMenu
  };
})(window);
