// Archivo: maq-modulos-registry.js
// Utilidad: Combina módulos fijos (sistema) con módulos detectados automáticamente.
(function (window) {
  "use strict";

  // Prefijo para los módulos de sistema (Gestion, Stats, etc.)
  // Asumimos ".." igual que antes.
  var base = "..";

  // 1. Módulos del SISTEMA (Fijos, no se escanean)
  var staticModules = [
    { id: "modulo_reporte",   nombre: "Reporte",      ruta: base + "/Reportes/repor.html",          categoria: "sistema", habilitado: true },
    { id: "carga_excel",      nombre: "Carga",        ruta: base + "/Gestion/Excel/excel.html",     categoria: "sistema", habilitado: true },
    { id: "tabla_principal",  nombre: "Tabla",        ruta: base + "/Gestion/Tabla/tabla.html",     categoria: "sistema", habilitado: true },
    { id: "ficha_estudiante", nombre: "Ficha",        ruta: base + "/Ficha/ficha.html",             categoria: "sistema", habilitado: true },
    { id: "stat_main",        nombre: "Estadísticas", ruta: base + "/Stats/stats.html",             categoria: "sistema", habilitado: true }
  ];

  // 2. Obtener datos automáticos (si existen)
  var autoData = window.MAQ_AUTODATA || [];

  function procesarAutoData(datos) {
    var lista = [];
    if (!Array.isArray(datos)) return lista;

    datos.forEach(function(item) {
      if (item.tipo === 'modulo') {
        lista.push({
          id: item.id,
          nombre: item.etiqueta,
          ruta: item.ruta,
          categoria: 'auto',
          habilitado: true
        });
      } else if (item.tipo === 'grupo' && Array.isArray(item.hijos)) {
        item.hijos.forEach(function(hijo) {
          lista.push({
            id: hijo.id,
            nombre: hijo.etiqueta,
            ruta: hijo.ruta,
            categoria: 'auto',
            habilitado: true
          });
        });
      }
    });
    return lista;
  }

  // 3. Combinar todo
  var dynamicModules = procesarAutoData(autoData);
  var allModules = staticModules.concat(dynamicModules);

  // 4. API Pública
  var registry = {
    lista: allModules,
    
    buscarPorId: function (id) {
      for (var i = 0; i < allModules.length; i++) {
        if (allModules[i].id === id) return allModules[i];
      }
      return null;
    },

    registrarExtras: function (extras) {
      // Compatibilidad con código antiguo
      if (Array.isArray(extras)) {
        // Podríamos agregarlos a allModules si fuera necesario
      }
    },

    listarHabilitados: function () {
      return allModules.filter(function(m) { return m.habilitado; });
    }
  };

  window.MAQ_MODULOS_REGISTRY = registry;
})(window);
